"""

from html.parser import HTMLParser

class NewsParser(HTMLParser):
    """
    –ü–∞—Ä—Å–µ—Ä HTML –¥–ª—è –∏–∑–≤–ª–µ—á–µ–Ω–∏—è –Ω–æ–≤–æ—Å—Ç–µ–π —Å —Å–∞–π—Ç–∞ itProger
    """
    
    def __init__(self):
        super().__init__()
        self.articles = []
        self.current_article = {}
        self.in_article = False
        self.in_title = False
        self.in_description = False
        self.title_collected = False
        self.description_collected = False
        self.current_link = None
        
    def handle_starttag(self, tag, attrs):
        """
        –û–±—Ä–∞–±–∞—Ç—ã–≤–∞–µ—Ç –æ—Ç–∫—Ä—ã–≤–∞—é—â–∏–µ —Ç–µ–≥–∏ HTML
        """
        attrs_dict = dict(attrs)
        
        # –£–ø—Ä–æ—â–µ–Ω–Ω—ã–π –ø–∞—Ä—Å–∏–Ω–≥ –¥–ª—è –ø–æ–∏—Å–∫–∞ –Ω–æ–≤–æ—Å—Ç–µ–π
        if tag == 'div':
            classes = attrs_dict.get('class', '')
            if 'article' in classes or 'news' in classes or 'post' in classes:
                self.in_article = True
                self.current_article = {}
                self.title_collected = False
                self.description_collected = False
                self.current_link = None
            
        elif tag == 'a' and self.in_article:
            # –ò—â–µ–º —Å—Å—ã–ª–∫—É –Ω–∞ —Å—Ç–∞—Ç—å—é
            href = attrs_dict.get('href', '')
            if href and ('/news/' in href or '/article/' in href):
                if not href.startswith('http'):
                    href = 'https://itproger.com' + href
                self.current_article['link'] = href
            
        elif tag in ['h3', 'h4', 'h2', 'span', 'div'] and self.in_article and not self.title_collected:
            self.in_title = True
                
        elif tag in ['p', 'span', 'div'] and self.in_article and self.title_collected and not self.description_collected:
            self.in_description = True

    def handle_endtag(self, tag):
        """
        –û–±—Ä–∞–±–∞—Ç—ã–≤–∞–µ—Ç –∑–∞–∫—Ä—ã–≤–∞—é—â–∏–µ —Ç–µ–≥–∏ HTML
        """

def fetch_url(url):
    """
    –ó–∞–≥—Ä—É–∂–∞–µ—Ç HTML —Å—Ç—Ä–∞–Ω–∏—Ü—É
    
    Args:
        url (str): URL –¥–ª—è –∑–∞–≥—Ä—É–∑–∫–∏
    
    Returns:
        str: HTML —Å–æ–¥–µ—Ä–∂–∏–º–æ–µ —Å—Ç—Ä–∞–Ω–∏—Ü—ã –∏–ª–∏ None –ø—Ä–∏ –æ—à–∏–±–∫–µ
    """
    try:
        parsed = urllib.parse.urlparse(url)
        conn = http.client.HTTPSConnection(parsed.netloc)
        path = parsed.path
        if parsed.query:
            path += '?' + parsed.query
            
        headers = {
            'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36',
            'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,image/webp,*/*;q=0.8',
        }
        
        conn.request("GET", path, headers=headers)
        response = conn.getresponse()
        
        if response.status == 200:
            content = response.read()
            return content.decode('utf-8', errors='ignore')
        else:
            print(f"‚ùå HTTP –æ—à–∏–±–∫–∞ {response.status} –¥–ª—è {url}")
            return None
    except Exception as e:
        print(f"‚ùå –û—à–∏–±–∫–∞ –∑–∞–≥—Ä—É–∑–∫–∏ URL {url}: {e}")
        return None

–ú–æ–¥—É–ª—å –¥–ª—è –æ–±—Ä–∞–±–æ—Ç–∫–∏ —Å–æ–æ–±—â–µ–Ω–∏–π –∏ callback'–æ–≤
"""

from database import save_user, log_request, get_user_favorites, get_user_stats
from database import add_to_favorites, remove_from_favorites, is_in_favorites
from parser import get_itproger_news
from keyboards import format_article_response, create_navigation_keyboard, create_articles_list_keyboard
from telegram_api import send_message, edit_message
from config import user_states

def handle_callback_query(callback_query):
    """
    –û–±—Ä–∞–±–∞—Ç—ã–≤–∞–µ—Ç callback –æ—Ç –∏–Ω–ª–∞–π–Ω –∫–Ω–æ–ø–æ–∫
    
    Args:
        callback_query (dict): –î–∞–Ω–Ω—ã–µ callback –æ—Ç Telegram
    """
    try:
        user_id = callback_query["from"]["id"]
        chat_id = callback_query["message"]["chat"]["id"]
        message_id = callback_query["message"]["message_id"]
        data = callback_query["data"]
        
        print(f"üîÑ –û–±—Ä–∞–±–æ—Ç–∫–∞ callback: {data} –æ—Ç –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è {user_id}")
        
        # –°–æ—Ö—Ä–∞–Ω—è–µ–º –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
        save_user(callback_query["from"])
        
        # –õ–æ–≥–∏—Ä—É–µ–º –∑–∞–ø—Ä–æ—Å
        log_request(user_id, f"callback:{data}")
        
        # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä—É–µ–º —Å–æ—Å—Ç–æ—è–Ω–∏–µ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è, –µ—Å–ª–∏ –Ω—É–∂–Ω–æ
        if user_id not in user_states:
            user_states[user_id] = {"current_page": 0, "articles": [], "current_article_index": 0}
        
        user_state = user_states[user_id]
        articles = user_state.get("articles", [])
        
        if not articles:
            # –ï—Å–ª–∏ –Ω–µ—Ç —Å—Ç–∞—Ç–µ–π, –∑–∞–≥—Ä—É–∂–∞–µ–º –∑–∞–Ω–æ–≤–æ
            articles = get_itproger_news()
            user_state["articles"] = articles
        
        if data.startswith("nav_"):
            # –ù–∞–≤–∏–≥–∞—Ü–∏—è –ø–æ —Å—Ç–∞—Ç—å—è–º
            article_index = int(data.split("_")[1]) - 1
            
            if 0 <= article_index < len(articles):
                user_state["current_article_index"] = article_index
                article = articles[article_index]
                
                response = format_article_response(article, article_index + 1, len(articles), user_id)
                keyboard = create_navigation_keyboard(article_index + 1, len(articles), user_id, article)
                
                edit_message(chat_id, message_id, response, keyboard)
            else:
                send_message(chat_id, "‚ùå –û—à–∏–±–∫–∞ –Ω–∞–≤–∏–≥–∞—Ü–∏–∏")
        
        elif data.startswith("article_"):
            # –ü—Ä–æ—Å–º–æ—Ç—Ä –∫–æ–Ω–∫—Ä–µ—Ç–Ω–æ–π —Å—Ç–∞—Ç—å–∏
            article_index = int(data.split("_")[1]) - 1
            
            if 0 <= article_index < len(articles):
                user_state["current_article_index"] = article_index
                article = articles[article_index]
                
                response = format_article_response(article, article_index + 1, len(articles), user_id)
                keyboard = create_navigation_keyboard(article_index + 1, len(articles), user_id, article)
                
                edit_message(chat_id, message_id, response, keyboard)
            else:
                send_message(chat_id, "‚ùå –°—Ç–∞—Ç—å—è –Ω–µ –Ω–∞–π–¥–µ–Ω–∞")
        
        elif data.startswith("page_"):
            # –ù–∞–≤–∏–≥–∞—Ü–∏—è –ø–æ —Å—Ç—Ä–∞–Ω–∏—Ü–∞–º —Å–ø–∏—Å–∫–∞
            page = int(data.split("_")[1])
            total_pages = (len(articles) + 4) // 5  # 5 —Å—Ç–∞—Ç–µ–π –Ω–∞ —Å—Ç—Ä–∞–Ω–∏—Ü—É
            
            if 0 <= page < total_pages:
                user_state["current_page"] = page
                
                response = f"üì∞ <b>–°—Ç—Ä–∞–Ω–∏—Ü–∞ {page + 1}</b>\n–í—ã–±–µ—Ä–∏—Ç–µ –Ω–æ–≤–æ—Å—Ç—å –¥–ª—è –ø–æ–¥—Ä–æ–±–Ω–æ–≥–æ –ø—Ä–æ—Å–º–æ—Ç—Ä–∞:"
                keyboard = create_articles_list_keyboard(articles, page)
                
                edit_message(chat_id, message_id, response, keyboard)
            else:
                send_message(chat_id, "‚ùå –°—Ç—Ä–∞–Ω–∏—Ü–∞ –Ω–µ –Ω–∞–π–¥–µ–Ω–∞")
        
        elif data == "back_to_list":
            # –í–æ–∑–≤—Ä–∞—Ç –∫ —Å–ø–∏—Å–∫—É –Ω–æ–≤–æ—Å—Ç–µ–π
            page = user_state.get("current_page", 0)
            
            response = f"üì∞ <b>–°—Ç—Ä–∞–Ω–∏—Ü–∞ {page + 1}</b>\n–í—ã–±–µ—Ä–∏—Ç–µ –Ω–æ–≤–æ—Å—Ç—å –¥–ª—è –ø–æ–¥—Ä–æ–±–Ω–æ–≥–æ –ø—Ä–æ—Å–º–æ—Ç—Ä–∞:"
            keyboard = create_articles_list_keyboard(articles, page)
            
            edit_message(chat_id, message_id, response, keyboard)
        
        elif data == "refresh_news":
            # –û–±–Ω–æ–≤–ª–µ–Ω–∏–µ –Ω–æ–≤–æ—Å—Ç–µ–π
            new_articles = get_itproger_news()
            user_state["articles"] = new_articles
            user_state["current_page"] = 0
            
            response = f"üì∞ <b>–ù–∞–π–¥–µ–Ω–æ {len(new_articles)} –Ω–æ–≤–æ—Å—Ç–µ–π</b>\n–í—ã–±–µ—Ä–∏—Ç–µ –Ω–æ–≤–æ—Å—Ç—å –¥–ª—è –ø–æ–¥—Ä–æ–±–Ω–æ–≥–æ –ø—Ä–æ—Å–º–æ—Ç—Ä–∞:"
            keyboard = create_articles_list_keyboard(new_articles, 0)
            
            edit_message(chat_id, message_id, response, keyboard)
        
        elif data.startswith("add_fav_"):
            # –î–æ–±–∞–≤–ª–µ–Ω–∏–µ –≤ –∏–∑–±—Ä–∞–Ω–Ω–æ–µ
            parts = data.split("_")
            article_index = int(parts[2]) - 1 if len(parts) >= 3 else int(parts[1]) - 1
            
            if 0 <= article_index < len(articles):
                article = articles[article_index]
                print(f"üîÑ –î–æ–±–∞–≤–ª–µ–Ω–∏–µ –≤ –∏–∑–±—Ä–∞–Ω–Ω–æ–µ: {article['title']}")
                
                success = add_to_favorites(user_id, article['title'], article.get('description', ''), article.get('link', 'https://itproger.com/news'))
                
                if success:
                    # –û–±–Ω–æ–≤–ª—è–µ–º —Å–æ–æ–±—â–µ–Ω–∏–µ —Å –Ω–æ–≤—ã–º —Å—Ç–∞—Ç—É—Å–æ–º
                    response = format_article_response(article, article_index + 1, len(articles), user_id)
                    keyboard = create_navigation_keyboard(article_index + 1, len(articles), user_id, article)
                    
                    edit_message(chat_id, message_id, response, keyboard)
                    # –û—Ç–ø—Ä–∞–≤–ª—è–µ–º –ø–æ–¥—Ç–≤–µ—Ä–∂–¥–µ–Ω–∏–µ
                    send_message(chat_id, f"‚úÖ –ù–æ–≤–æ—Å—Ç—å –¥–æ–±–∞–≤–ª–µ–Ω–∞ –≤ –∏–∑–±—Ä–∞–Ω–Ω–æ–µ!\n\n<b>{article['title']}</b>")
                    
                    # –ü—Ä–æ–≤–µ—Ä—è–µ–º, —á—Ç–æ –¥–µ–π—Å—Ç–≤–∏—Ç–µ–ª—å–Ω–æ —Å–æ—Ö—Ä–∞–Ω–∏–ª–æ—Å—å –≤ –ë–î
                    if is_in_favorites(user_id, article['title']):
                        print(f"‚úÖ –ü–æ–¥—Ç–≤–µ—Ä–∂–¥–µ–Ω–∏–µ: –Ω–æ–≤–æ—Å—Ç—å '{article['title']}' —Å–æ—Ö—Ä–∞–Ω–µ–Ω–∞ –≤ –ë–î")
                    else:
                        print(f"‚ùå –û—à–∏–±–∫–∞: –Ω–æ–≤–æ—Å—Ç—å –Ω–µ —Å–æ—Ö—Ä–∞–Ω–∏–ª–∞—Å—å –≤ –ë–î")
                else:
                    send_message(chat_id, "‚ùå –û—à–∏–±–∫–∞ –¥–æ–±–∞–≤–ª–µ–Ω–∏—è –≤ –∏–∑–±—Ä–∞–Ω–Ω–æ–µ")
            else:
                send_message(chat_id, "‚ùå –°—Ç–∞—Ç—å—è –Ω–µ –Ω–∞–π–¥–µ–Ω–∞")
        
        elif data.startswith("remove_fav_"):
            # –£–¥–∞–ª–µ–Ω–∏–µ –∏–∑ –∏–∑–±—Ä–∞–Ω–Ω–æ–≥–æ
            parts = data.split("_")
            article_index = int(parts[2]) - 1 if len(parts) >= 3 else int(parts[1]) - 1
            
            if 0 <= article_index < len(articles):
                article = articles[article_index]
                print(f"üîÑ –£–¥–∞–ª–µ–Ω–∏–µ –∏–∑ –∏–∑–±—Ä–∞–Ω–Ω–æ–≥–æ: {article['title']}")
                
                success = remove_from_favorites(user_id, article['title'])
                
                if success:
                    # –û–±–Ω–æ–≤–ª—è–µ–º —Å–æ–æ–±—â–µ–Ω–∏–µ —Å –Ω–æ–≤—ã–º —Å—Ç–∞—Ç—É—Å–æ–º
                    response = format_article_response(article, article_index + 1, len(articles), user_id)
                    keyboard = create_navigation_keyboard(article_index + 1, len(articles), user_id, article)
                    
                    edit_message(chat_id, message_id, response, keyboard)
                    # –û—Ç–ø—Ä–∞–≤–ª—è–µ–º –ø–æ–¥—Ç–≤–µ—Ä–∂–¥–µ–Ω–∏–µ
                    send_message(chat_id, f"‚úÖ –ù–æ–≤–æ—Å—Ç—å —É–¥–∞–ª–µ–Ω–∞ –∏–∑ –∏–∑–±—Ä–∞–Ω–Ω–æ–≥–æ!\n\n<b>{article['title']}</b>")
                else:
                    send_message(chat_id, "‚ùå –û—à–∏–±–∫–∞ —É–¥–∞–ª–µ–Ω–∏—è –∏–∑ –∏–∑–±—Ä–∞–Ω–Ω–æ–≥–æ")
            else:
                send_message(chat_id, "‚ùå –°—Ç–∞—Ç—å—è –Ω–µ –Ω–∞–π–¥–µ–Ω–∞")
        
        else:
            send_message(chat_id, "‚ùå –ù–µ–∏–∑–≤–µ—Å—Ç–Ω–∞—è –∫–æ–º–∞–Ω–¥–∞")
            
    except Exception as e:
        print(f"‚ùå –û—à–∏–±–∫–∞ –æ–±—Ä–∞–±–æ—Ç–∫–∏ callback: {e}")
        send_message(chat_id, "‚ùå –ü—Ä–æ–∏–∑–æ—à–ª–∞ –æ—à–∏–±–∫–∞ –ø—Ä–∏ –æ–±—Ä–∞–±–æ—Ç–∫–µ –∑–∞–ø—Ä–æ—Å–∞")

def handle_message(message):
    """
    –û–±—Ä–∞–±–∞—Ç—ã–≤–∞–µ—Ç —Ç–µ–∫—Å—Ç–æ–≤—ã–µ —Å–æ–æ–±—â–µ–Ω–∏—è –æ—Ç –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
    
    Args:
        message (dict): –°–æ–æ–±—â–µ–Ω–∏–µ –æ—Ç Telegram
    """
    chat_id = message["chat"]["id"]
    text = message["text"].strip()
    user_info = message["from"]
    
    print(f"üì® –°–æ–æ–±—â–µ–Ω–∏–µ –æ—Ç {user_info.get('first_name', 'Unknown')} ({chat_id}): {text}")
    
    # –°–æ—Ö—Ä–∞–Ω—è–µ–º –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
    save_user(user_info)
    
    # –õ–æ–≥–∏—Ä—É–µ–º –∑–∞–ø—Ä–æ—Å
    log_request(user_info["id"], text)
    
    if text.startswith("http"):
        # –û–±—Ä–∞–±–æ—Ç–∫–∞ URL
        if "itproger.com/news" in text:
            from parser import fetch_url, parse_news, save_news_to_cache
            html = fetch_url(text)
            if html:
                articles = parse_news(html)
                if articles:
                    save_news_to_cache(articles)
                    # –°–æ—Ö—Ä–∞–Ω—è–µ–º —Å—Ç–∞—Ç—å–∏ –≤ —Å–æ—Å—Ç–æ—è–Ω–∏–∏ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
                    user_id = user_info["id"]
                    if user_id not in user_states:
                        user_states[user_id] = {"current_page": 0, "articles": [], "current_article_index": 0}
                    user_states[user_id]["articles"] = articles
                    user_states[user_id]["current_page"] = 0
                    
                    response = f"üì∞ <b>–ù–∞–π–¥–µ–Ω–æ {len(articles)} –Ω–æ–≤–æ—Å—Ç–µ–π</b>\n–í—ã–±–µ—Ä–∏—Ç–µ –Ω–æ–≤–æ—Å—Ç—å –¥–ª—è –ø–æ–¥—Ä–æ–±–Ω–æ–≥–æ –ø—Ä–æ—Å–º–æ—Ç—Ä–∞:"
                    keyboard = create_articles_list_keyboard(articles, 0)
                    send_message(chat_id, response, keyboard)
                else:
                    send_message(chat_id, "‚ùå –ù–µ —É–¥–∞–ª–æ—Å—å –Ω–∞–π—Ç–∏ —Å—Ç–∞—Ç—å–∏ –Ω–∞ —Å—Ç—Ä–∞–Ω–∏—Ü–µ")
            else:
                send_message(chat_id, "‚ùå –û—à–∏–±–∫–∞ –ø—Ä–∏ –∑–∞–≥—Ä—É–∑–∫–µ —Å—Ç—Ä–∞–Ω–∏—Ü—ã. –ü—Ä–æ–≤–µ—Ä—å—Ç–µ URL.")
        else:
            send_message(chat_id, "üîó –Ø —Å–ø–µ—Ü–∏–∞–ª–∏–∑–∏—Ä—É—é—Å—å –Ω–∞ –ø–∞—Ä—Å–∏–Ω–≥–µ itproger.com/news\n–û—Ç–ø—Ä–∞–≤—å—Ç–µ —Å—Å—ã–ª–∫—É –Ω–∞ —ç—Ç–æ—Ç —Å–∞–π—Ç –∏–ª–∏ –∫–æ–º–∞–Ω–¥—É /news")
            
    elif text == "/start":
        response = """üëã <b>–ü—Ä–∏–≤–µ—Ç! –Ø –±–æ—Ç –¥–ª—è –ø–∞—Ä—Å–∏–Ω–≥–∞ –Ω–æ–≤–æ—Å—Ç–µ–π itProger</b>

üìã <b>–î–æ—Å—Ç—É–ø–Ω—ã–µ –∫–æ–º–∞–Ω–¥—ã:</b>
/news - –ü–æ–ª—É—á–∏—Ç—å –ø–æ—Å–ª–µ–¥–Ω–∏–µ –Ω–æ–≤–æ—Å—Ç–∏ —Å –Ω–∞–≤–∏–≥–∞—Ü–∏–µ–π
/favorites - –ú–æ–∏ –∏–∑–±—Ä–∞–Ω–Ω—ã–µ –Ω–æ–≤–æ—Å—Ç–∏
/stats - –ú–æ—è —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞
/help - –ü–æ–∫–∞–∑–∞—Ç—å —Å–ø—Ä–∞–≤–∫—É

üîó <b>–ò–ª–∏ –ø—Ä–æ—Å—Ç–æ –æ—Ç–ø—Ä–∞–≤—å—Ç–µ —Å—Å—ã–ª–∫—É:</b>
https://itproger.com/news"""
        send_message(chat_id, response)
        
    elif text == "/news":
        articles = get_itproger_news()
        if articles:
            # –°–æ—Ö—Ä–∞–Ω—è–µ–º —Å—Ç–∞—Ç—å–∏ –≤ —Å–æ—Å—Ç–æ—è–Ω–∏–∏ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
            user_id = user_info["id"]
            if user_id not in user_states:
                user_states[user_id] = {"current_page": 0, "articles": [], "current_article_index": 0}
            user_states[user_id]["articles"] = articles
            user_states[user_id]["current_page"] = 0
            
            response = f"üì∞ <b>–ù–∞–π–¥–µ–Ω–æ {len(articles)} –Ω–æ–≤–æ—Å—Ç–µ–π</b>\n–í—ã–±–µ—Ä–∏—Ç–µ –Ω–æ–≤–æ—Å—Ç—å –¥–ª—è –ø–æ–¥—Ä–æ–±–Ω–æ–≥–æ –ø—Ä–æ—Å–º–æ—Ç—Ä–∞:"
            keyboard = create_articles_list_keyboard(articles, 0)
            send_message(chat_id, response, keyboard)
        else:
            send_message(chat_id, "‚ùå –ù–µ —É–¥–∞–ª–æ—Å—å –∑–∞–≥—Ä—É–∑–∏—Ç—å –Ω–æ–≤–æ—Å—Ç–∏. –ü–æ–ø—Ä–æ–±—É–π—Ç–µ –ø–æ–∑–∂–µ.")
        
    elif text == "/favorites":
        favorites = get_user_favorites(user_info["id"])
        if favorites:
            # –°–æ—Ö—Ä–∞–Ω—è–µ–º –∏–∑–±—Ä–∞–Ω–Ω—ã–µ –≤ —Å–æ—Å—Ç–æ—è–Ω–∏–∏ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
            user_id = user_info["id"]
            if user_id not in user_states:
                user_states[user_id] = {"current_page": 0, "articles": [], "current_article_index": 0}
            
            # –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ–º –∏–∑–±—Ä–∞–Ω–Ω—ã–µ –≤ —Ñ–æ—Ä–º–∞—Ç —Å—Ç–∞—Ç–µ–π
            fav_articles = [{"title": fav['title'], "description": fav['description'], "link": fav.get('link', 'https://itproger.com/news')} for fav in favorites]
            user_states[user_id]["articles"] = fav_articles
            user_states[user_id]["current_page"] = 0
            
            response = f"‚≠ê <b>–í–∞—à–∏ –∏–∑–±—Ä–∞–Ω–Ω—ã–µ –Ω–æ–≤–æ—Å—Ç–∏ ({len(favorites)})</b>\n–í—ã–±–µ—Ä–∏—Ç–µ –Ω–æ–≤–æ—Å—Ç—å –¥–ª—è –ø—Ä–æ—Å–º–æ—Ç—Ä–∞:"
            keyboard = create_articles_list_keyboard(fav_articles, 0)
            send_message(chat_id, response, keyboard)
        else:
            send_message(chat_id, "üìù –£ –≤–∞—Å –ø–æ–∫–∞ –Ω–µ—Ç –∏–∑–±—Ä–∞–Ω–Ω—ã—Ö –Ω–æ–≤–æ—Å—Ç–µ–π.\n–ò—Å–ø–æ–ª—å–∑—É–π—Ç–µ –∫–æ–º–∞–Ω–¥—É /news —á—Ç–æ–±—ã –¥–æ–±–∞–≤–∏—Ç—å.")
            
    elif text == "/stats":
        stats = get_user_stats(user_info["id"])
        response = f"""üìä <b>–í–∞—à–∞ —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞:</b>

üì® –ó–∞–ø—Ä–æ—Å–æ–≤ –∫ –±–æ—Ç—É: {stats['request_count']}
‚≠ê –ò–∑–±—Ä–∞–Ω–Ω—ã—Ö –Ω–æ–≤–æ—Å—Ç–µ–π: {stats['favorites_count']}
üïê –ü–æ—Å–ª–µ–¥–Ω—è—è –∞–∫—Ç–∏–≤–Ω–æ—Å—Ç—å: {stats['last_activity'][:16]}"""
        send_message(chat_id, response)
        
    elif text == "/help":
        response = """‚ÑπÔ∏è <b>–°–ø—Ä–∞–≤–∫–∞ –ø–æ –±–æ—Ç—É</b>

–≠—Ç–æ—Ç –±–æ—Ç –ø–∞—Ä—Å–∏—Ç –∑–∞–≥–æ–ª–æ–≤–∫–∏ –∏ –æ–ø–∏—Å–∞–Ω–∏—è —Å—Ç–∞—Ç–µ–π —Å —Å–∞–π—Ç–∞ itproger.com/news

<b>–ö–æ–º–∞–Ω–¥—ã:</b>
/start - –ù–∞—á–∞—Ç—å —Ä–∞–±–æ—Ç—É
/news - –ü–æ–ª—É—á–∏—Ç—å —Å–≤–µ–∂–∏–µ –Ω–æ–≤–æ—Å—Ç–∏ —Å –Ω–∞–≤–∏–≥–∞—Ü–∏–µ–π
/favorites - –ò–∑–±—Ä–∞–Ω–Ω—ã–µ –Ω–æ–≤–æ—Å—Ç–∏
/stats - –°—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞
/help - –≠—Ç–∞ —Å–ø—Ä–∞–≤–∫–∞
